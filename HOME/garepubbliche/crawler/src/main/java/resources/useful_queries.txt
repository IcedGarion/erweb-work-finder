
#percentuale di oggetti non nulli (crawler)
select tot, notNull, (notNull/tot)*100 as percentuale_notNull from 
(
	select 
	( select count(*) from erweb.bando ) as tot,
	( select count(*) from erweb.bando where oggetto != '') as notNull 
) as t
